```bash
PAYPAL_CLIENT_ID=...
PAYPAL_CLIENT_SECRET=...

CLOUDINARY_CLOUD_NAME=...
CLOUDINARY_API_KEY=...
CLOUDINARY_API_SECRET=...

MONGO_URL=...

PORT="5000"

CLIENT_BASE_URL="http://localhost:5173"

NODE_ENV=development
```
